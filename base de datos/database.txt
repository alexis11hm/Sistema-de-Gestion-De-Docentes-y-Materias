drop database if exists sistematec;
create database sistematec;
use sistematec;

create table personal(id int auto_increment not null primary key,
			titulo varchar(30) not null,
			nombre varchar(30) not null,
			paterno varchar(30) not null,
			materno varchar(30) not null,
			sexo varchar(10) not null,
			rfc varchar(30) not null,
			curp varchar(30) not null,
			foto varchar(30) not null,
			correo varchar(30) not null,
			usuario varchar(30) not null,
			password varchar(30) not null,
			tipo int(1) not null default 0);

create table departamento(id int auto_increment not null primary key,
			nombre varchar(30) not null,
			abreviatura varchar(30) not null);

create table puesto(id int auto_increment not null primary key,
		nombre varchar(30) not null);

create table puestodepartamento(id int auto_increment not null primary key,
		personal int not null,
		departamento int not null,
		puesto int not null);

create table materia(id int auto_increment not null primary key,
		nombre varchar(30) not null);

create table docentemateria(id int auto_increment not null primary key,
		materia int not null,
		puestodepartamento int not null);

create table calificacion(id int auto_increment not null primary key,
			unidad int(1) not null,
			documento varchar(30) not null,
			docentemateria int not null);
